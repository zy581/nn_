# ============================================================
# 特殊安装说明
# ============================================================
# 
# 1. 基础安装（PyPI 依赖）：
#    pip install -r requirements.txt
#
# 2. MuJoCo 安装：
#    - 获取许可证：https://mujoco.org/
#    - 设置 MUJOCO_KEY_PATH 环境变量
#    - 或使用免费版：https://github.com/deepmind/mujoco
#
# 3. Isaac Lab 安装（可选）：
#    git clone https://github.com/isaac-sim/IsaacLab.git
#    cd IsaacLab
#    ./isaaclab.sh --install
#
# 4. ROS 安装（可选，用于 main.py）：
#    参考：http://wiki.ros.org/ROS/Installation
#    需要 rospy 和 std_msgs
#
# 5. UITB 框架安装：
#    pip install -e .  # 从项目根目录安装
#
# 6. 无头渲染（GPU服务器）：
#    export MUJOCO_GL=egl
#    export DISPLAY=:0
#
# 7. BeatSVR 模型需要额外下载：
#    - SMPL 模型文件（data/smpl/）
#    - 预训练策略文件（output/）
#    - AMASS 运动数据（sample_data/）
#
